function updateSnippet(pkZ32, tsMicros) {
  const theme = $('theme-select').value;
  const useStaging = $('use-staging').checked;
  const snippet = generateSnippet(pkZ32, tsMicros, theme, useStaging);
  $('code-snippet').textContent = snippet;
  return snippet;
}

async function showWizard(pubky, session) {
  step1View.style.display  = 'none';
  wizardView.style.display = 'block';

  const pk    = session.info.publicKey;
  const pkStr = pk.toString();
  const pkZ32 = pk.z32();

  // Generate timestamp in microseconds
  const tsMicros = (BigInt(Date.now()) * 1000n).toString();

  // Render user info (without profile first, then update)
  renderUserInfo(pkStr, pkZ32, null);
  try {
    const profile = await pubky.publicStorage.getJson(
      `${pkStr}/pub/pubky.app/profile.json`
    );
    if (profile) renderUserInfo(pkStr, pkZ32, profile);
  } catch { /* ignore */ }

  // Show step 2
  goToStep(2);

  // Generate initial snippet
  let currentSnippet = updateSnippet(pkZ32, tsMicros);

  // Update snippet when options change
  $('theme-select').addEventListener('change', () => {
    currentSnippet = updateSnippet(pkZ32, tsMicros);
    $('copy-status').textContent = '';
  });
  $('use-staging').addEventListener('change', () => {
    currentSnippet = updateSnippet(pkZ32, tsMicros);
    $('copy-status').textContent = '';
  });

  // Copy button
  $('copy-btn').addEventListener('click', async () => {
    if (!currentSnippet) return;
    try {
      await navigator.clipboard.writeText(currentSnippet);
      $('copy-status').textContent = '✓ Copied to clipboard!';
    } catch {
      $('copy-status').textContent = 'Copy failed — please select and copy manually.';
    }
  });

  // Next button → Step 3
  $('step2-next-btn').addEventListener('click', () => {
    goToStep(3);
  });

  // Sign out → reload page (will show QR again since no session caching)
  $('wizard-signout').addEventListener('click', () => {
    location.reload();
  });
}

async function main() {
  const pubky = new Pubky();

  // Always start fresh QR auth flow (no session caching)
  let flow;
  try {
    flow = pubky.startAuthFlow(AUTH_CAPS, AuthFlowKind.signin());
  } catch (e) {
    showError(errorBox, 'Failed to start auth flow: ' + e.message);
    return;
  }

  // Render QR code
  try {
    await QRCode.toCanvas(
      $('qr-canvas'),
      flow.authorizationUrl,
      { width: 256, margin: 1, color: { dark: '#0f172a', light: '#ffffff' } }
    );
    statusEl.textContent = 'Waiting for approval…';
  } catch (e) {
    showError(errorBox, 'Failed to render QR: ' + e.message);
    return;
  }

  // Wait for user to approve
  try {
    const session = await flow.awaitApproval();
    // Do NOT store session - page refresh will ask for QR login again
    await showWizard(pubky, session);
  } catch (e) {
    showError(errorBox, 'Auth failed: ' + e.message);
    statusEl.textContent = 'Authentication failed.';
  }
}

main().catch(e => showError(errorBox, 'Unexpected error: ' + e.message));
